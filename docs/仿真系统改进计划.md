## 一、架构优化（参考 OASIS）

### 1.1 异步并行架构升级
**当前状态**：使用 `Promise.all` 实现基本并行
**改进方向**：
- [ ] 实现事件驱动的消息队列系统
- [ ] 创建信息通道（Information Channel）处理 agent 请求
- [ ] 添加 UUID 追踪系统确保请求可追溯
- [ ] 实现线程安全的响应字典

```typescript
// world-slice/src/engine/message-queue.ts
export class MessageQueue {
  private queue: Array<{ id: string; request: any; timestamp: number }> = []
  private responses: Map<string, any> = new Map()
  
  async enqueue(request: any): Promise<string> {
    const id = crypto.randomUUID()
    this.queue.push({ id, request, timestamp: Date.now() })
    return id
  }
  
  async dequeue(): Promise<{ id: string; request: any } | null> {
    return this.queue.shift() || null
  }
  
  setResponse(id: string, response: any): void {
    this.responses.set(id, response)
  }
  
  async waitForResponse(id: string, timeout = 30000): Promise<any> {
    const start = Date.now()
    while (!this.responses.has(id)) {
      if (Date.now() - start > timeout) {
        throw new Error(`Request ${id} timed out`)
      }
      await new Promise(resolve => setTimeout(resolve, 100))
    }
    const response = this.responses.get(id)
    this.responses.delete(id)
    return response
  }
}
```

### 1.2 推荐系统（RecSys）
**当前状态**：无推荐系统，agents 看到所有信息
**改进方向**：
- [ ] 实现基于兴趣的内容推荐
- [ ] 实现基于热度的内容推荐
- [ ] 使用嵌入模型（如 OpenAI Embeddings）计算相似度
- [ ] 缓存推荐结果到数据库

```typescript
// world-slice/src/engine/recommendation-system.ts
export class RecommendationSystem {
  async recommendEvents(
    agent: PersonalAgentState,
    world: WorldSlice,
    topK: number = 10
  ): Promise<WorldEvent[]> {
    // 1. 基于兴趣的推荐
    const interestBasedEvents = await this.getInterestBasedEvents(agent, world)
    
    // 2. 基于热度的推荐
    const hotEvents = await this.getHotEvents(world)
    
    // 3. 社交网络推荐（关注的人的动态）
    const socialEvents = await this.getSocialNetworkEvents(agent, world)
    
    // 4. 混合推荐
    return this.mergeRecommendations(
      interestBasedEvents,
      hotEvents,
      socialEvents,
      topK
    )
  }
  
  private async getInterestBasedEvents(
    agent: PersonalAgentState,
    world: WorldSlice
  ): Promise<WorldEvent[]> {
    // 使用 agent 的 expertise、goals、core_belief 计算兴趣向量
    const agentInterests = this.computeInterestVector(agent)
    
    // 计算每个事件与 agent 兴趣的相似度
    const scoredEvents = world.events.map(event => ({
      event,
      score: this.cosineSimilarity(
        agentInterests,
        this.computeEventVector(event)
      )
    }))
    
    return scoredEvents
      .sort((a, b) => b.score - a.score)
      .map(item => item.event)
  }
  
  private async getHotEvents(world: WorldSlice): Promise<WorldEvent[]> {
    // 基于事件类型、影响范围、时间新鲜度计算热度
    return world.events
      .map(event => ({
        event,
        hotScore: this.calculateHotScore(event, world.tick)
      }))
      .sort((a, b) => b.hotScore - a.hotScore)
      .map(item => item.event)
  }
  
  private calculateHotScore(event: WorldEvent, currentTick: number): number {
    const recency = Math.log10(Math.max(1, currentTick - (event.tick || 0)))
    const importance = this.getEventImportance(event.type)
    return importance / (recency + 1)
  }
}
```

```typescript
    typical_wake_hour: number     // 通常起床时间（0-23）
  }
}

// world-slice/src/engine/time-engine.ts
export class TimeEngine {
  shouldActivateAgent(agent: PersonalAgentState, currentHour: number): boolean {
    const probability = agent.activity_pattern[currentHour]
    return Math.random() < probability
  }
  
  getActiveAgents(agents: PersonalAgentState[], world: WorldSlice): PersonalAgentState[] {
    const currentHour = new Date(world.time).getHours()
    return agents.filter(agent => this.shouldActivateAgent(agent, currentHour))
  }
}
```

```typescript
// world-slice/src/engine/knowledge-graph.ts
export type KnowledgeNode = {
  id: string
  type: 'agent' | 'location' | 'organization' | 'event' | 'concept'
  properties: Record<string, any>
  embedding?: number[]
}

export type KnowledgeEdge = {
  id: string
  source: string  // node id
  target: string  // node id
  relation: string  // 'knows', 'located_in', 'works_for', 'caused_by', etc.
  weight: number
  properties: Record<string, any>
}

export class KnowledgeGraph {
  private nodes: Map<string, KnowledgeNode> = new Map()
  private edges: Map<string, KnowledgeEdge> = new Map()
  
  addNode(node: KnowledgeNode): void {
    this.nodes.set(node.id, node)
  }
  
  addEdge(edge: KnowledgeEdge): void {
    this.edges.set(edge.id, edge)
  }
  
  // 查询与某个 agent 相关的所有信息
  getAgentContext(agentId: string, depth: number = 2): {
    nodes: KnowledgeNode[]
    edges: KnowledgeEdge[]
  } {
    const visited = new Set<string>()
    const resultNodes: KnowledgeNode[] = []
    const resultEdges: KnowledgeEdge[] = []
    
    const traverse = (nodeId: string, currentDepth: number) => {
      if (currentDepth > depth || visited.has(nodeId)) return
      visited.add(nodeId)
      
      const node = this.nodes.get(nodeId)
      if (node) resultNodes.push(node)
      
      // 找到所有相关边
      for (const edge of this.edges.values()) {
        if (edge.source === nodeId || edge.target === nodeId) {
          resultEdges.push(edge)
          const nextNodeId = edge.source === nodeId ? edge.target : edge.source
          traverse(nextNodeId, currentDepth + 1)
        }
      }
    }
    
    traverse(agentId, 0)
    return { nodes: resultNodes, edges: resultEdges }
  }
  
  // 从世界状态构建知识图谱
  static async buildFromWorld(world: WorldSlice): Promise<KnowledgeGraph> {
    const graph = new KnowledgeGraph()
    
    // 添加 agents 作为节点
    for (const agent of world.agents.npcs) {
      graph.addNode({
        id: agent.genetics.seed,
        type: 'agent',
        properties: {
          name: agent.identity.name,
          occupation: agent.occupation,
          role: agent.role,
          expertise: agent.expertise,
          goals: agent.goals,
        }
      })
      
      // 添加关系边
      for (const [otherAgentName, relationValue] of Object.entries(agent.relations)) {
        const otherAgent = world.agents.npcs.find(a => a.identity.name === otherAgentName)
        if (otherAgent) {
          graph.addEdge({
            id: `${agent.genetics.seed}-knows-${otherAgent.genetics.seed}`,
            source: agent.genetics.seed,
            target: otherAgent.genetics.seed,
            relation: relationValue > 0 ? 'likes' : 'dislikes',
            weight: Math.abs(relationValue),
            properties: {}
          })
        }
      }
    }
    
    // 添加剧情作为节点
    for (const plot of world.plots) {
      graph.addNode({
        id: plot.id,
        type: 'event',
        properties: {
          title: plot.title,
          type: plot.type,
          status: plot.status,
        }
      })
      
      // 连接剧情与 agents
      for (const protagonistSeed of plot.protagonists) {
        graph.addEdge({
          id: `${plot.id}-protagonist-${protagonistSeed}`,
          source: plot.id,
          target: protagonistSeed,
          relation: 'has_protagonist',
          weight: 1.0,
          properties: {}
        })
      }
    }
    
    return graph
  }
}
```

### 2.2 记忆系统增强
**当前状态**：简单的短期/长期记忆列表
**改进方向**：
- [ ] 基于知识图谱的记忆检索
- [ ] 记忆重要性自动评估
- [ ] 记忆关联和推理

```typescript
// world-slice/src/engine/memory-system.ts
export class EnhancedMemorySystem {
  constructor(
    private knowledgeGraph: KnowledgeGraph
  ) {}
  
  async retrieveRelevantMemories(
    agent: PersonalAgentState,
    context: string,
    topK: number = 5
  ): Promise<Memory[]> {
    // 1. 从知识图谱获取相关上下文
    const graphContext = this.knowledgeGraph.getAgentContext(agent.genetics.seed)
    
    // 2. 结合短期和长期记忆
    const allMemories = [...agent.memory_short, ...agent.memory_long]
    
    // 3. 计算每个记忆与当前上下文的相关性
    const scoredMemories = allMemories.map(memory => ({
      memory,
      score: this.calculateRelevance(memory, context, graphContext)
    }))
    
    // 4. 返回最相关的记忆
    return scoredMemories
      .sort((a, b) => b.score - a.score)
      .slice(0, topK)
      .map(item => item.memory)
  }
  
  private calculateRelevance(
    memory: Memory,
    context: string,
    graphContext: { nodes: KnowledgeNode[]; edges: KnowledgeEdge[] }
  ): number {
    // 综合考虑：
    // 1. 记忆重要性
    // 2. 情感权重
    // 3. 检索强度
    // 4. 与知识图谱的关联度
    // 5. 时间衰减
    
    const importanceScore = memory.importance
    const emotionalScore = memory.emotional_weight
    const retrievalScore = memory.retrieval_strength
    const graphScore = this.calculateGraphRelevance(memory, graphContext)
    const timeDecay = Math.exp(-memory.decay_rate * 0.1)
    
    return (
      importanceScore * 0.3 +
      emotionalScore * 0.2 +
      retrievalScore * 0.2 +
      graphScore * 0.2 +
      timeDecay * 0.1
    )
  }
}
```

```typescript
// world-slice/src/engine/report-agent.ts
export class ReportAgent {
  async generatePredictionReport(
    world: WorldSlice,
    question: string
  ): Promise<PredictionReport> {
    // 1. 分析世界当前状态
    const currentState = this.analyzeWorldState(world)
    
    // 2. 识别关键趋势
    const trends = this.identifyTrends(world)
    
    // 3. 预测未来发展
    const predictions = await this.makePredictions(world, question, trends)
    
    // 4. 生成报告
    return {
      question,
      current_state: currentState,
      key_trends: trends,
      predictions,
      confidence_level: this.calculateConfidence(predictions),
      generated_at: new Date().toISOString(),
    }
  }
  
  private analyzeWorldState(world: WorldSlice): WorldStateAnalysis {
    return {
      total_agents: world.agents.npcs.length,
      active_plots: world.plots.filter(p => p.status === 'active').length,
      social_pressure: this.calculateAveragePressure(world),
      dominant_emotions: this.getDominantEmotions(world),
      key_relationships: this.analyzeRelationships(world),
    }
  }
  
  private identifyTrends(world: WorldSlice): Trend[] {
    // 分析事件历史，识别趋势
    const recentEvents = world.events.slice(-100)
    
    // 情绪趋势
    const emotionTrend = this.analyzeEmotionTrend(world)
    
    // 关系趋势
    const relationshipTrend = this.analyzeRelationshipTrend(world)
    
    // 剧情进展趋势
    const plotTrend = this.analyzePlotTrend(world)
    
    return [emotionTrend, relationshipTrend, plotTrend]
  }
}
```

```typescript
// world-slice/src/engine/counterfactual-engine.ts
export class CounterfactualEngine {
  async simulateWhatIf(
    baseWorld: WorldSlice,
    intervention: Intervention,
    steps: number = 10
  ): Promise<CounterfactualResult> {
    // 1. 复制世界状态
    const altWorld = this.cloneWorld(baseWorld)
    
    // 2. 应用干预
    this.applyIntervention(altWorld, intervention)
    
    // 3. 运行仿真
    const timeline: WorldSlice[] = [altWorld]
    for (let i = 0; i < steps; i++) {
      const nextWorld = await runWorldTick(timeline[timeline.length - 1])
      timeline.push(nextWorld)
    }
    
    // 4. 对比分析
    const comparison = this.compareTimelines(baseWorld, timeline)
    
    return {
      intervention,
      timeline,
      comparison,
      divergence_points: this.identifyDivergencePoints(baseWorld, timeline),
    }
  }
}

export type Intervention = {
  type: 'add_agent' | 'remove_agent' | 'change_relationship' | 'inject_event' | 'modify_plot'
  target: string
  parameters: Record<string, any>
}
```
